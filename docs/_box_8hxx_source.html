577 Real squared_interior_distance(Box const & b, Point & p1, Point & p2) const
578 {
579 if (this->intersects(b)) {return 0.0;}
580 for (Integer i{0}; i < N; ++i) {
581 if (this->m_min[i] > b.m_max[i]) {p1[i] = b.m_max[i]; p2[i] = this->m_min[i];}
582 else if (b.m_min[i] > this->m_max[i]) {p1[i] = this->m_max[i]; p2[i] = b.m_min[i];}
583 else {p1[i] = p2[i] = 0.5*(std::min(this->m_max[i], b.m_max[i]) + std::max(this->m_min[i], b.m_min[i]));}
584 }
585 return (p2 - p1).squaredNorm();
