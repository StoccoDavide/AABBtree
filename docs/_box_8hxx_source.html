403 Side which_side(Real const x, Real const tol, Integer const dim) const {
404 Real const mi { m_min(dim) };
405 Real const ma { m_max(dim) };
406 bool const on_left { ma < x + tol };
407 bool const on_right { x - tol < mi };
408 if (on_left && on_right) {
409 return (mi+ma)/2 < x ? Side::LEFT : Side::RIGHT;
410 }
411 if (on_left) return Side::LEFT;
412 if (on_right) return Side::RIGHT;
413 return Side::INSIDE;
414 }

478 {

481 if (m_min[i] > p[i]) {Real aux{m_min[i] - p[i]}; dist2 += aux*aux;}

582 Real squared_interior_distance(Box const & b) const {
583 if (this->intersects(b)) return 0;
584 Real dist2{0.0};
585 for (Integer i{0}; i < N; ++i) {
586 Real aux{ m_min[i] - b.m_max[i]};
587 if (aux < 0.0) {
588 aux = b.m_min[i] - m_max[i];
589 if (aux < 0.0) {continue;}
590 }
591 dist2 += aux*aux;
592 }
593 return dist2;
594 }

605 Real squared_interior_distance(Box const & b, Point & p1, Point & p2) const {
606 if (this->intersects(b)) {return 0.0;}
607 for (Integer i{0}; i < N; ++ i) {
608 if (m_min[i] > b.m_max[i]) {p1[i] = b.m_max[i]; p2[i] = m_min[i];}
609 else if (b.m_min[i] > m_max[i]) {p1[i] = m_max[i]; p2[i] = b.m_min[i];}
610 else {
611 p1[i] = p2[i] = 0.5*(std::min(m_max[i], b.m_max[i]) + std::max(m_min[i], b.m_min[i]));
612 }
613 }
614 return (p2 - p1).squaredNorm();
615 }
