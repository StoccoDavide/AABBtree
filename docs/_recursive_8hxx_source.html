88 void build_impl(BoxUniquePtrList const & boxes) {
89 // Check if the input vector is empty
90 if (boxes.empty() ) return;
91
92 // Check if the input vector contains only one box
93 if (boxes.size() == static_cast<Integer>(1)) {
94 this->m_box.emplace_back(boxes.front());
95 return;
96 }
97
98 // Compute the bounding box of the input vector
99 this->m_box = std::make_unique<Box>();
100 this->m_box->extend(boxes);
101
102 // Cut on longest edge of the bounding box
103 Integer n_cut{this->m_box->max_dimension()};
104 Real x_min{this->m_box->min(n_cut)};
105 Real x_max{this->m_box->max(n_cut)};
106 Real x_cut{0.5*(x_max + x_min)};
107
108 // Collect the boxes that have a size similar to the current box
109 this->m_objects.clear();
110 for (Box const & box : boxes)
111 {
112 Real x_mid{0.5*(box->min(n_cut) + box->max(n_cut))};
113 if (x_mid > x_cut) {this->m_objects.emplace_back(box);}
114 }
115
116 // Split the boxes into left and right
117 BoxUniquePtrList left_boxes;
118 BoxUniquePtrList right_boxes;
119 for (auto const & b : boxes ) {
120 Real x_mid{(b.min(n_cut) + b.max(n_cut))*static_cast<Real>(0.5)};
121 if (x_mid > x_cut) {left_boxes.emplace_back(b);}
122 else {right_boxes.emplace_back(b);}
123 }
124
125 // Build the left and right trees
126 TreePtr left_tree{std::make_unique<Recursive>()};
127 TreePtr right_tree{std::make_unique<Recursive>()};
128 left_tree->build(left_boxes);
129 right_tree->build(right_boxes);
130 if (!left_tree->is_empty()) {m_children.first(left_tree);}
131 if (!right_tree->is_empty()) {m_children.second(right_tree);}
132 }
