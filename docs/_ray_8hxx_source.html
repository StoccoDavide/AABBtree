226 bool intersects(Box<Real, N> const & b, Real tol = DUMMY_TOL) const
227 {
228 if (b.contains(this->m_origin)) {return true;}
229 Point const & b_min{b.min()};
230 Point const & b_max{b.max()};
231 Vector t_min, t_max; t_min.setConstant(-MAX), t_max.setConstant(MAX);
232 for (Integer i{0}; i < N; ++i) {
233 if (std::abs(this->m_direction[i]) > tol) {
234 t_min[i] = (b_min[i] - this->m_origin[i])/this->m_direction[i];
235 t_max[i] = (b_max[i] - this->m_origin[i])/this->m_direction[i];
236 if (t_min[i] > t_max[i]) {std::swap(t_min[i], t_max[i]);}
237 } else if (this->m_origin[i] < b_min[i] || this->m_origin[i] > b_max[i]) {
238 return false;
239 }
240 }
241 Real t_entry{t_min.maxCoeff()};
242 Real t_exit{t_max.minCoeff()};
243 return t_entry <= t_exit && t_exit >= -tol;
244 }

343 Real squared_interior_distance(Box<Real, N> const & b, Point & p1, Point & p2, Real tol = DUMMY_TOL) const
344 {
345 if (b.contains(this->m_origin)) {return 0.0;}
346 Point const & b_min{b.min()};
347 Point const & b_max{b.max()};
348
349 // Compute intersection parameters
350 Vector t_min, t_max; t_min.setConstant(-MAX); t_max.setConstant(MAX);
351 for (Integer i{0}; i < N; ++i) {
352 if (std::abs(this->m_direction[i]) > tol) {
353 t_min[i] = (b_min[i] - this->m_origin[i])/this->m_direction[i];
354 t_max[i] = (b_max[i] - this->m_origin[i])/this->m_direction[i];
355 if (t_min[i] > t_max[i]) {std::swap(t_min[i], t_max[i]);}
356 } else if (this->m_origin[i] < b_min[i] || this->m_origin[i] > b_max[i]) {
357 // Ray is parallel and outside the box and non-intersecting
358 b.interior_distance(this->m_origin, p2);
359 Vector sides{b_max - b_min};
360 this->distance(p2, p1);
361 for (Integer j{0}; j < N; ++j){
362 if (j == i) {continue;}
363 if (this->m_direction[j] > 0.0) {p1[j] += 0.5*sides[j]; p2[j] += 0.5*sides[j];}
364 else {p1[j] -= 0.5*sides[j]; p2[j] -= 0.5*sides[j];}
365 }
366 return (p2 - this->m_origin).norm();
367 }
368 }
369
370 // Compute global intersection range
371 Real t_entry{t_min.maxCoeff()}, t_exit{t_max.minCoeff()};
372 if (t_entry <= t_exit && t_exit >= 0) {return 0.0;}
373
374 // Compute closest point if no intersection
375 for (Integer i{0}; i < N; ++i) {
376 if (this->m_origin[i] < b_min[i]) {p2[i] = b_min[i];}
377 else if (this->m_origin[i] > b_max[i]) {p2[i] = b_max[i];}
378 else {p2[i] = this->m_origin[i];}
379 }
380
381 // Project closest point onto the ray
382 Vector v(p2 - this->m_origin);
383 Real t_proj{v.dot(this->m_direction)/this->m_direction.squaredNorm()};
384 if (t_proj < 0.0) {p1 = this->m_origin; return v.norm();}
385
386 p1 = this->m_origin + t_proj * this->m_direction;;
387 return (p2 - p1).squaredNorm();
388 }
