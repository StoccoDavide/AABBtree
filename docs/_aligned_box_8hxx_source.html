356 Real squared_distance(const Point & p) const
357 {
358 if (this->contains(p)) {return static_cast<Real>(0.0);}
359 Real dist2{0.0}, aux;
360 for (Integer i{0}; i < N; ++i)
361 {
362 if (this->m_min[i] > p[i]) {
363 aux = this->m_min[i] - p[i];
364 dist2 += aux*aux;
365 } else if (p[i] > this->m_max[i]) {
366 aux = p[i] - this->m_max[i];
367 dist2 += aux*aux;
368 }
369 }
370 return dist2;
371 }

447 Real squared_interior_distance(const AlignedBox & b, Point & p_min, Point & p_max) const
448 {
449 if (this->intersects(b)) {return static_cast<Real>(0.0);}
450 for (Integer i{0}; i < N; ++i)
451 {
452 if (this->m_min[i] > b.m_max[i]) {
453 p_min[i] = b.m_max[i];
454 p_max[i] = this->m_min[i];
455 } else if (b.m_min[i] > this->m_max[i]) {
456 p_min[i] = this->m_max[i];
457 p_max[i] = b.m_min[i];
458 } else {
459 p_min[i] = p_max[i] =
460 (std::min(this->m_max[i], b.m_max[i]) + std::max(this->m_min[i], b.m_min[i]))/
461 static_cast<Real>(2.0);
462 }
463 }
464 return (p_max - p_min).squaredNorm();
465 }
