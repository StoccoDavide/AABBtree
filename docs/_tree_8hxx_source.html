274 void build(std::unique_ptr<BoxUniquePtrList> boxes)
275 {
276
277 // Collect the original object boxes
278 this->m_boxes = std::move(boxes);
279
280 // Clear tree structure
281 Integer num_boxes{this->m_boxes->size()};
282 this->m_tree_structure.clear();
283 this->m_tree_structure.reserve(static_cast<size_t>(2*num_boxes));
284
285 // Setup the root node
286 AABBnode root;
287 root.parent = -1;
288 root.child_l = -1;
289 root.child_r = -1;
290 root.box_ptr = 0;
291 root.box_num = 0;
292
293 // Setup the boxes map and compute the root box
294 Integer depth{std::ceil(std::log2(num_boxes))};
295 root.box.set_empty();
296 this->m_tree_boxes_map.reserve(2*depth);
297 for (BoxUniquePtr const & box : *this->m_boxes) {
298 root.box.extend(*box);
299 this->m_tree_boxes_map.emplace_back(root.box_num++);
300 }
301 this->m_tree_structure.emplace_back(root);
302
303 // Setup the stack
304 this->m_stack.clear();
305 this->m_stack.reserve(2*num_boxes + 1);
306 this->m_stack.emplace_back(0);
307
308 // Setup the dump counter (axis to try)
309 Integer dump{0};
310
311 // Main loop that divide the nodes iteratively until all constraints satisfied
312 while (!this->m_stack.empty())
313 {
314 // Pop the node from stack
315 Integer const id{this->m_stack.back()}; this->m_stack.pop_back();
316
317 // Get the node
318 AABBnode & node{this->m_tree_structure[id]};
319
320 // If the node has less than the maximum number of objects, skip it
321 if (node.box_num < this->m_max_nodal_objects) {continue;}
322
323 // Compute the separation line and tolerance
324 Vector sizes;
325 Eigen::Vector<Integer, N> sorting;
326 node.box.sort_axes_length(sizes, sorting);
327 Integer axis{sorting[dump]};
328 Real separation_line{node.box.baricenter(axis)};
329 Real separation_tolerance{sizes[axis] * this->m_separation_ratio_tolerance};
330
331 // Separate short and long boxes and compute short boxes baricenter
332 Integer n_long{0};
333 Integer n_left{0};
334 Integer n_right{0};
335 Integer id_ini{node.box_ptr};
336 Integer id_end{node.box_ptr + node.box_num};
337 Real baricenter {0.0};
338 while (id_ini < id_end) {
339 Box const & box_id{*(*this->m_boxes)[this->m_tree_boxes_map[id_ini]]};
340 Integer side{static_cast<Integer>(box_id.which_side(separation_line, separation_tolerance, axis))};
341 switch (side) {
342 case static_cast<Integer>(Box::Side::LEFT): ++n_left; // Left boxes are moved to the end
343 std::swap(this->m_tree_boxes_map[id_ini], this->m_tree_boxes_map[--id_end]); break;
344 case static_cast<Integer>(Box::Side::RIGHT): ++n_right; // Right boxes are moved to the end
345 std::swap(this->m_tree_boxes_map[id_ini], this->m_tree_boxes_map[--id_end]); break;
346 default: ++n_long; ++id_ini;
347 }
348 baricenter += box_id.max()[axis] + box_id.min()[axis];
349 }
350 baricenter /= 2.0*node.box_num;
351
352 // Check if the leafs are balanced, if not try to separate boxes on a new separation line
353 if (n_long > this->m_max_nodal_objects ||
354 std::min(n_left, n_right) < this->m_balance_ratio_tolerance * std::max(n_left, n_right) ) {
355
356 // Perform coputation only if baricenter is well separated from the previous separation line
357 if (std::abs(baricenter - separation_line) > separation_tolerance) {
358 n_long = n_left = n_right = 0;
359 id_ini = node.box_ptr;
360 id_end = node.box_ptr + node.box_num;
361 while (id_ini < id_end) {
362 Box const & box_id{*(*this->m_boxes)[this->m_tree_boxes_map[id_ini]]};
363 Integer side{static_cast<Integer>(box_id.which_side(baricenter, separation_tolerance, axis))};
364 switch (side) {
365 case static_cast<Integer>(Box::Side::LEFT): ++n_left;
366 std::swap(this->m_tree_boxes_map[id_ini], this->m_tree_boxes_map[--id_end]); break;
367 case static_cast<Integer>(Box::Side::RIGHT): ++n_right;
368 std::swap(this->m_tree_boxes_map[id_ini], this->m_tree_boxes_map[--id_end]); break;
369 default: ++n_long; ++id_ini;
370 }
371 }
372 }
373 }
374
375 // If the left and right children are yet not balanced, dump the splitting axis
376 if (this->m_dumping_mode && n_long > this->m_max_nodal_objects) {
377 if (std::min(n_left, n_right) < this->m_balance_ratio_tolerance * std::max(n_left, n_right)) {
378 if (dump < N-1) {++dump; this->m_stack.push_back(id); {continue;}}
379 }}
380
381 // Reset the dump counter
382 dump = 0;
383
384 // Separate the left and right boxes
385 n_left = n_right = 0;
386 id_ini = node.box_ptr + n_long;
387 id_end = node.box_ptr + node.box_num;
388 while (id_ini < id_end) {
389 Integer ipos{this->m_tree_boxes_map[id_ini]};
390 if ((*this->m_boxes)[ipos]->baricenter(axis) < separation_line) {
391 ++id_ini; ++n_left; // In right position do nothing
392 } else {
393 --id_end; ++n_right; // In right position swap the current box with the last one
394 std::swap(this->m_tree_boxes_map[id_ini], this->m_tree_boxes_map[id_end]);
395 }
396 }
397
398 // If the left and right children have few boxes, they are leafs
399 if (n_left <= this->m_max_nodal_objects && n_right <= this->m_max_nodal_objects) {continue;}
400
401 // Set the left and right children indexes
402 node.child_l = static_cast<Integer>(this->size() + 0);
403 node.child_r = static_cast<Integer>(this->size() + 1);
404
405 // Finalize the root node setup (left and right children)
406 AABBnode node_l;
407 node_l.parent = id; // Current node
408 node_l.child_l = -1; // Default (leaf)
409 node_l.child_r = -1; // Default (leaf)
410 node_l.box_num = n_left;
411
412 AABBnode node_r;
413 node_r.parent = id; // Current node
414 node_r.child_l = -1; // Default (leaf)
415 node_r.child_r = -1; // Default (leaf)
416 node_r.box_num = n_right;
417
418 // Compute the bounding box of the long boxes, and left and right children
419 Integer j{node.box_ptr};
420 node.box_num = n_long;
421 node.box_long.set_empty();
422 for (Integer i{0}; i < n_long; ++i) node.box_long.extend(*(*this->m_boxes)[m_tree_boxes_map[j++]]);
423 node_l.box.set_empty(); node_l.box_ptr = j;
424 for (Integer i{0}; i < n_left; ++i) node_l.box.extend(*(*this->m_boxes)[m_tree_boxes_map[j++]]);
425 node_r.box.set_empty(); node_r.box_ptr = j;
426 for (Integer i{0}; i < n_right; ++i) node_r.box.extend(*(*this->m_boxes)[m_tree_boxes_map[j++]]);
427
428 // Push nodes on tree structure
429 this->m_tree_structure.emplace_back(node_l);
430 this->m_tree_structure.emplace_back(node_r);
431
432 // Push children on stack
433 this->m_stack.emplace_back(node.child_l);
434 this->m_stack.emplace_back(node.child_r);
435 }
436 }

446 template <typename Object>
447 bool intersect(Object const & obj, IndexSet & candidates) const
448 {
449 // Reset statistics
450 this->m_check_counter = 0;
451
452 // Return if the tree is empty
453 if (this->is_empty()) {return false;}
454
455 // Collect the original object boxes
456 BoxUniquePtrList const & boxes{*this->m_boxes};
457
458 // Setup the stack
459 this->m_stack.clear();
460 this->m_stack.reserve(2*this->size() + 1);
461 this->m_stack.emplace_back(0);
462
463 // Main loop that checks the intersection iteratively
464 candidates.clear();
465 while (!this->m_stack.empty())
466 {
467 // Pop the node from stack
468 Integer const id{this->m_stack.back()}; this->m_stack.pop_back();
469
470 // Get the node
471 AABBnode const & node{this->m_tree_structure[id]};
472
473 // If the object is not intersects the box, skip the node
474 ++this->m_check_counter;
475 if (!node.box.intersects(obj)) {continue;}
476
477 // Intersect the object with the long boxes on the node
478 Integer const id_ini{node.box_ptr};
479 Integer const id_end{node.box_ptr + node.box_num};
480 for (Integer i{id_ini}; i < id_end; ++i) {
481 Integer const pos{this->m_tree_boxes_map[i]};
482 ++this->m_check_counter;
483 if (boxes[pos]->intersects(obj)) {candidates.insert(pos);}
484 }
485
486 // Push children on the stack if thay are not leafs
487 if (node.child_l > 0) {this->m_stack.emplace_back(node.child_l);}
488 if (node.child_r > 0) {this->m_stack.emplace_back(node.child_r);}
489 }
490
491 // Return true if the object intersects the tree
492 return !candidates.empty();
493 }

501 bool intersect(Tree const & tree, IndexMap & candidates) const
502 {
503 // Reset statistics
504 this->m_check_counter = 0;
505
506 // Return if the tree is empty
507 if (this->is_empty() || tree.is_empty()) {return false;}
508
509 // Collect the original object boxes
510 BoxUniquePtrList const & boxes_1{*this->m_boxes};
511 BoxUniquePtrList const & boxes_2{*tree.m_boxes};
512
513 // Setup the stack
514 this->m_stack.clear();
515 this->m_stack.reserve(this->size() + tree.size() + 2);
516 this->m_stack.emplace_back(0);
517 this->m_stack.emplace_back(0);
518
519 // Main loop that checks the intersection iteratively
520 candidates.clear();
521 while (!this->m_stack.empty())
522 {
523 // Pop the node from stack (reversed order)
524 Integer const id_2{this->m_stack.back()}; this->m_stack.pop_back();
525 Integer const id_1{this->m_stack.back()}; this->m_stack.pop_back();
526
527 // Get the node
528 AABBnode const & node_1{this->m_tree_structure[id_1]};
529 AABBnode const & node_2{tree.m_tree_structure[id_2]};
530
531 // If the boxes are not intersecting, skip the nodes
532 ++this->m_check_counter;
533 if (!node_1.box.intersects(node_2.box)) {continue;}
534
535 // Intersect the long boxes on the nodes
536 if (node_1.box_num > 0 && node_2.box_num > 0) {
537 Integer const id_1_ini{node_1.box_ptr};
538 Integer const id_1_end{node_1.box_ptr + node_1.box_num};
539 Integer const id_2_ini{node_2.box_ptr};
540 Integer const id_2_end{node_2.box_ptr + node_2.box_num};
541 for (Integer i{id_1_ini}; i < id_1_end; ++i) {
542 Integer const pos_1{this->m_tree_boxes_map[i]};
543 for (Integer j{id_2_ini}; j < id_2_end; ++j) {
544 Integer const pos_2{tree.m_tree_boxes_map[j]};
545 ++this->m_check_counter;
546 if (boxes_1[pos_1]->intersects(*boxes_2[pos_2])) {candidates[pos_1].insert(pos_2);}
547 }
548 }
549 }
550
551 // Push children of both treees on the stack if thay are not leafs
552 auto push_on_stack = [this](Integer const left_1, Integer const left_2) {
553 if (left_1 >= 0 && left_2 >= 0)
554 {this->m_stack.emplace_back(left_1); this->m_stack.emplace_back(left_2);}
555 };
556 push_on_stack(node_1.child_l, node_2.child_l);
557 push_on_stack(node_1.child_r, node_2.child_r);
558 push_on_stack(id_1, node_2.child_l);
559 push_on_stack(id_1, node_2.child_r);
560 push_on_stack(node_1.child_l, id_2);
561 push_on_stack(node_1.child_r, id_2);
562 }
563
564 // Return true if the trees intersect
565 return !candidates.empty();
566 }

573 bool self_intersect(IndexSet & candidates) const
574 {
575 IndexMap candidates_map;
576 bool intersects{this->intersect(*this, candidates_map)};
577 candidates.clear();
578 for (const auto & [key, values] : candidates_map) {
579 for (int value : values) {candidates.emplace(key); candidates.emplace(value);}
580 }
581 return intersects;

592 template <typename Object>
593 Real distance(Object const & obj, IndexSet & candidates) const
594 {
595 // Reset statistics
596 this->m_check_counter = 0;
597
598 // Return a negative value if the tree is empty
599 if (this->is_empty() || this->is_empty()) {return -1.0;}
600
601 // Collect the original object boxes
602 BoxUniquePtrList const & boxes{*this->m_boxes};
603
604 // Setup the stack
605 this->m_stack.clear();
606 this->m_stack.reserve(2*this->size() + 1);
607 this->m_stack.emplace_back(0);
608 this->m_stack.emplace_back(0);
609
610 // Main loop that checks the intersection iteratively
611 Real distance{std::numeric_limits<Real>::max()};
612 candidates.clear();
613 while (!this->m_stack.empty())
614 {
615 // Pop the node from stack
616 Integer const id{this->m_stack.back()}; this->m_stack.pop_back();
617
618 // Get the node
619 AABBnode const & node {this->m_tree_structure[id]};
620
621 // Compute the distance between the object and the bounding box
622 ++this->m_check_counter;
623 Real tmp_distance{node.box.interior_distance(obj)};
624
625 // If the distance is greater than the temporary minimum distance, skip the node
626 if (tmp_distance > distance) {continue;}
627
628 // Compute the distance between the object and the long boxes on the node
629 Integer const id_ini{node.box_ptr};
630 Integer const id_end{node.box_ptr + node.box_num};
631 for (Integer i{id_ini}; i < id_end; ++i) {
632 Integer const pos{this->m_tree_boxes_map[i]};
633 ++this->m_check_counter;
634 tmp_distance = boxes[pos]->interior_distance(obj);
635 if (tmp_distance < distance) {
636 candidates.clear(); candidates.insert(pos); distance = tmp_distance;
637 } else if (tmp_distance == distance) {
638 candidates.insert(pos);
639 }
640 }
641
642 // Push children on the stack if thay are not leafs
643 if (node.child_l > 0) {this->m_stack.emplace_back(node.child_l);}
644 if (node.child_r > 0) {this->m_stack.emplace_back(node.child_r);}
645 }
646
647 // Return the distance between the point and the tree
648 return distance;
649 }

657 Real distance(Tree const & tree, IndexMap & candidates) const
658 {
659 // Reset statistics
660 this->m_check_counter = 0;
661
662 // Return if the tree is empty
663 if (this->is_empty() || tree.is_empty()) {return -1.0;}
664
665 // Collect the original object boxes
666 BoxUniquePtrList const & boxes_1{*this->m_boxes};
667 BoxUniquePtrList const & boxes_2{*tree.m_boxes};
668
669 // Setup the stack
670 this->m_stack.clear();
671 this->m_stack.reserve(this->size() + tree.size() + 2);
672 this->m_stack.emplace_back(0);
673 this->m_stack.emplace_back(0);
674
675 // Main loop that checks the intersection iteratively
676 Real distance{std::numeric_limits<Real>::max()};
677 candidates.clear();
678 while (!this->m_stack.empty())
679 {
680 // Pop the node from stack (reversed order)
681 Integer const id_2{this->m_stack.back()}; this->m_stack.pop_back();
682 Integer const id_1{this->m_stack.back()}; this->m_stack.pop_back();
683
684 // Get the node
685 AABBnode const & node_1{this->m_tree_structure[id_1]};
686 AABBnode const & node_2{tree.m_tree_structure[id_2]};
687
688 // Compute the distance between the bounding boxes
689 ++this->m_check_counter;
690 Real tmp_distance{node_1.box.interior_distance(node_2.box)};
691
692 // If the distance is greater than the temporary minimum distance, skip the nodes
693 if (tmp_distance > distance) {continue;}
694
695 // Compute the distance between the long boxes on the nodes
696 if (node_1.box_num > 0 && node_2.box_num > 0) {
697 Integer const id_1_ini{node_1.box_ptr};
698 Integer const id_1_end{node_1.box_ptr + node_1.box_num};
699 Integer const id_2_ini{node_2.box_ptr};
700 Integer const id_2_end{node_2.box_ptr + node_2.box_num};
701 for (Integer i{id_1_ini}; i < id_1_end; ++i) {
702 Integer const pos_1{this->m_tree_boxes_map[i]};
703 for (Integer j{id_2_ini}; j < id_2_end; ++j) {
704 Integer const pos_2{tree.m_tree_boxes_map[j]};
705 ++this->m_check_counter;
706 tmp_distance = boxes_1[pos_1]->interior_distance(*boxes_2[pos_2]);
707 if (tmp_distance < distance) {
