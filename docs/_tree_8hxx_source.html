245 void build(std::unique_ptr<BoxUniquePtrList> boxes)
246 {
247
248 // Collect the original object boxes
249 this->m_boxes = std::move(boxes);
250
251 // Clear tree structure
252 Integer num_boxes{this->m_boxes->size()};
253 this->m_tree_structure.clear();
254 this->m_tree_structure.reserve(20*num_boxes + 1);
255
256 // Setup the root node
257 Node root;
258 root.parent = -1;
259 root.child_l = -1;
260 root.child_r = -1;
261 root.box_ptr = 0;
262 root.box_num = 0;
263 root.box_tot_num = num_boxes;
264
265 // Setup the boxes map and compute the root box
266 Integer depth{std::ceil(std::log2(num_boxes))};
267 root.box.set_empty();
268 this->m_tree_boxes_map.reserve(2*depth);
269 for (BoxUniquePtr const & box : *this->m_boxes) {
270 root.box.extend(*box);
271 this->m_tree_boxes_map.emplace_back(root.box_num++);
272 }
273 this->m_tree_structure.emplace_back(root);
274
275 // Setup the stack
276 this->m_stack.clear();
277 this->m_stack.reserve(2*num_boxes + 1);
278 this->m_stack.emplace_back(0);
279
280 // Setup the dump counter (axis to try)
281 Integer dump{0};
282
283 // Main loop that divide the nodes iteratively until all constraints satisfied
284 while (!this->m_stack.empty())
285 {
286 // Pop the node from stack
287 Integer const id{this->m_stack.back()}; this->m_stack.pop_back();
288
289 // Get the node
290 Node & node{this->m_tree_structure[id]};
291
292 // If the node has less than the maximum number of objects, skip it
293 if (node.box_num < this->m_max_nodal_objects) {continue;}
294
295 // Compute the separation line and tolerance
296 Vector sizes;
297 Eigen::Vector<Integer, N> sorting;
298 node.box.sort_axes_length(sizes, sorting);
299 Integer axis{sorting[dump]};
300 Real separation_line{node.box.baricenter(axis)};
301 Real separation_tolerance{sizes[axis] * this->m_separation_ratio_tolerance};
302
303 // Separate short and long boxes and compute short boxes baricenter
304 Integer n_long{0};
305 Integer n_left{0};
306 Integer n_right{0};
307 Integer id_ini{node.box_ptr};
308 Integer id_end{node.box_ptr + node.box_num};
309 Real baricenter {0.0};
310 while (id_ini < id_end) {
311 Box const & box_id{*(*this->m_boxes)[this->m_tree_boxes_map[id_ini]]};
312 Integer side{static_cast<Integer>(box_id.which_side(separation_line, separation_tolerance, axis))};
313 switch (side) {
314 case static_cast<Integer>(Box::Side::LEFT): ++n_left; // Left boxes are moved to the end
315 std::swap(this->m_tree_boxes_map[id_ini], this->m_tree_boxes_map[--id_end]); break;
316 case static_cast<Integer>(Box::Side::RIGHT): ++n_right; // Right boxes are moved to the end
317 std::swap(this->m_tree_boxes_map[id_ini], this->m_tree_boxes_map[--id_end]); break;
318 default: ++n_long; ++id_ini;
319 }
320 baricenter += box_id.max()[axis] + box_id.min()[axis];
321 }
322 baricenter /= 2.0*node.box_num;
323
324 // Check if the leafs are balanced, if not try to separate boxes on a new separation line
325 if (n_long > this->m_max_nodal_objects ||
326 std::min(n_left, n_right) < this->m_balance_ratio_tolerance * std::max(n_left, n_right) ) {
327
328 // Perform coputation only if baricenter is well separated from the previous separation line
329 if (std::abs(baricenter - separation_line) > separation_tolerance) {
330 n_long = n_left = n_right = 0;
331 id_ini = node.box_ptr;
332 id_end = node.box_ptr + node.box_num;
333 while (id_ini < id_end) {
334 Box const & box_id{*(*this->m_boxes)[this->m_tree_boxes_map[id_ini]]};
335 Integer side{static_cast<Integer>(box_id.which_side(baricenter, separation_tolerance, axis))};
336 switch (side) {
337 case static_cast<Integer>(Box::Side::LEFT): ++n_left;
338 std::swap(this->m_tree_boxes_map[id_ini], this->m_tree_boxes_map[--id_end]); break;
339 case static_cast<Integer>(Box::Side::RIGHT): ++n_right;
340 std::swap(this->m_tree_boxes_map[id_ini], this->m_tree_boxes_map[--id_end]); break;
341 default: ++n_long; ++id_ini;
342 }
343 }
344 }
345 }
346
347 // If the left and right children are yet not balanced, dump the splitting axis
348 if (this->m_dumping_mode && n_long > this->m_max_nodal_objects) {
349 if (std::min(n_left, n_right) < this->m_balance_ratio_tolerance * std::max(n_left, n_right)) {
350 if (dump < N-1) {++dump; this->m_stack.push_back(id); {continue;}}
351 }}
352
353 // Reset the dump counter
354 dump = 0;
355
356 // Separate the left and right boxes
357 n_left = n_right = 0;
358 id_ini = node.box_ptr + n_long;
359 id_end = node.box_ptr + node.box_num;
360 while (id_ini < id_end) {
361 Integer ipos{this->m_tree_boxes_map[id_ini]};
362 if ((*this->m_boxes)[ipos]->baricenter(axis) < separation_line) {
363 ++id_ini; ++n_left; // In right position do nothing
364 } else {
365 --id_end; ++n_right; // In right position swap the current box with the last one
366 std::swap(this->m_tree_boxes_map[id_ini], this->m_tree_boxes_map[id_end]);
367 }
368 }
369
370 // If the left and right children have few boxes, they are leafs
371 if (n_left <= this->m_max_nodal_objects && n_right <= this->m_max_nodal_objects) {continue;}
372
373 // Set the left and right children indexes
374 node.child_l = static_cast<Integer>(this->size() + 0);
375 node.child_r = static_cast<Integer>(this->size() + 1);
376
377 // Finalize the root node setup (left and right children)
378 Node node_l;
379 node_l.parent = id; // Current node
380 node_l.child_l = -1; // Default (leaf)
381 node_l.child_r = -1; // Default (leaf)
382 node_l.box_num = n_left;
383 node_l.box_tot_num = n_left;
384
385 Node node_r;
386 node_r.parent = id; // Current node
387 node_r.child_l = -1; // Default (leaf)
388 node_r.child_r = -1; // Default (leaf)
389 node_r.box_num = n_right;
390 node_r.box_tot_num = n_right;
391
392 // Compute the bounding box of the long boxes, and left and right children
393 Integer j{node.box_ptr};
394 node.box_num = n_long;
395 node.box_long.set_empty();
396 for (Integer i{0}; i < n_long; ++i) {node.box_long.extend(*(*this->m_boxes)[this->m_tree_boxes_map[j++]]);}
397 node_l.box.set_empty(); node_l.box_ptr = j;
398 for (Integer i{0}; i < n_left; ++i) {node_l.box.extend(*(*this->m_boxes)[this->m_tree_boxes_map[j++]]);}
399 node_l.box_long = node_l.box;
400 node_r.box.set_empty(); node_r.box_ptr = j;
401 for (Integer i{0}; i < n_right; ++i) {node_r.box.extend(*(*this->m_boxes)[this->m_tree_boxes_map[j++]]);}
402 node_r.box_long = node_r.box;
403
404 // Push nodes on tree structure
405 this->m_tree_structure.emplace_back(node_l);
406 this->m_tree_structure.emplace_back(node_r);
407
408 // Push children on stack
409 this->m_stack.emplace_back(node.child_l);
410 this->m_stack.emplace_back(node.child_r);
411 }
412 }

422 template <typename Object>
423 bool intersect(Object const & obj, IndexSet & candidates) const
424 {
425 // Reset statistics
426 this->m_check_counter = 0;
427
428 // Return if the tree is empty
429 if (this->is_empty()) {return false;}
430
431 // Collect the original object boxes
432 BoxUniquePtrList const & boxes{*this->m_boxes};
433
434 // Setup the stack
435 this->m_stack.clear();
436 this->m_stack.reserve(2*this->size() + 1);
437 this->m_stack.emplace_back(0);
438
439 // Main loop that checks the intersection iteratively
440 candidates.clear();
441 while (!this->m_stack.empty())
442 {
443 // Pop the node from stack
444 Integer const id{this->m_stack.back()}; this->m_stack.pop_back();
445
446 // Get the node
447 Node const & node{this->m_tree_structure[id]};
448
449 // If the object is not intersects the box, skip the node
450 ++this->m_check_counter;
451 if (!node.box.intersects(obj)) {continue;}
452
453 // Intersect the object with the long boxes on the node
454 if (node.box_num > 0) {
455 ++this->m_check_counter;
456 if (node.box_long.intersects(obj)) {
457 Integer const id_ini{node.box_ptr};
458 Integer const id_end{node.box_ptr + node.box_num};
459 for (Integer i{id_ini}; i < id_end; ++i) {
460 Integer const pos{this->m_tree_boxes_map[i]};
461 ++this->m_check_counter;
462 if (boxes[pos]->intersects(obj)) {candidates.insert(pos);}
463 }
464 }
465 }
466
467 // Push children on the stack if thay are not leafs
468 if (node.child_l > 0) {this->m_stack.emplace_back(node.child_l);}
469 if (node.child_r > 0) {this->m_stack.emplace_back(node.child_r);}
470 }
471
472 // Return true if the object intersects the tree
473 return !candidates.empty();
474 }

482 bool intersect(Tree const & tree, IndexMap & candidates) const
483 {
484 // Reset statistics
485 this->m_check_counter = 0;
486
487 // Return if the tree is empty
488 if (this->is_empty() || tree.is_empty()) {return false;}
489
490 // Collect the original object boxes
491 BoxUniquePtrList const & boxes_1{*this->m_boxes};
492 BoxUniquePtrList const & boxes_2{*tree.m_boxes};
493
494 // Setup the stack
495 this->m_stack.clear();
496 this->m_stack.reserve(this->size() + tree.size() + 2);
497 this->m_stack.emplace_back(0);
498 this->m_stack.emplace_back(0);
499
500 // Negate the id of the node to distinguish the tree
501 auto negate = [] (Integer const id) {return -1-id;};
502
503 // Main loop that checks the intersection iteratively
504 candidates.clear();
505 while (!this->m_stack.empty())
506 {
507 // Pop the node from stack (reversed order)
508 Integer const id_s2{this->m_stack.back()}; this->m_stack.pop_back();
509 Integer const id_2{id_s2 >= 0 ? id_s2 : negate(id_s2)};
510 Integer const id_s1{this->m_stack.back()}; this->m_stack.pop_back();
511 Integer const id_1{id_s1 >= 0 ? id_s1 : negate(id_s1)};
512
513 // Get the node
514 Node const & node_1{this->m_tree_structure[id_1]};
515 Node const & node_2{tree.m_tree_structure[id_2]};
516
517 // If the boxes are not intersecting, skip the nodes
518 ++this->m_check_counter;
519 if (!node_1.box.intersects(node_2.box)) {continue;}
520
521 // Intersect the long boxes on the nodes
522 if (node_1.box_num > 0 && node_2.box_num > 0) {
523 ++this->m_check_counter;
524 if (node_1.box_long.intersects(node_2.box_long)) {
525 Integer const id_1_ini{node_1.box_ptr};
526 Integer const id_1_end{node_1.box_ptr + node_1.box_num};
527 Integer const id_2_ini{node_2.box_ptr};
528 Integer const id_2_end{node_2.box_ptr + node_2.box_num};
529 for (Integer i{id_1_ini}; i < id_1_end; ++i) {
530 Integer const pos_1{this->m_tree_boxes_map[i]};
531 for (Integer j{id_2_ini}; j < id_2_end; ++j) {
532 Integer const pos_2{tree.m_tree_boxes_map[j]};
533 ++this->m_check_counter;
534 if (boxes_1[pos_1]->intersects(*boxes_2[pos_2])) {candidates[pos_1].insert(pos_2);}
535 }
536 }
537 }
538 }
539
540 // Push children of both trees on the stack if thay are not leafs
541 auto stack_emplace_back = [this](Integer const node_1, Integer const node_2) {
542 this->m_stack.emplace_back(node_1); this->m_stack.emplace_back(node_2);
543 };
544 // FIXME if (id_s1 >= 0 && id_s2 >= 0) {
545 // if (node_1.box_tot_num >= node_2.box_tot_num) {
546 // stack_emplace_back(node_1.child_l, id_s2);
547 // stack_emplace_back(node_1.child_r, id_s2);
548 // if (node_1.box_num > 0) {stack_emplace_back(id_1, negate(id_2));}
549 // } else {
550 // stack_emplace_back(id_s1, node_2.child_l);
551 // stack_emplace_back(id_s1, node_2.child_r);
552 // if (node_2.box_num > 0) stack_emplace_back(negate(id_1), id_2);}
553 //} else
554 if (id_s1 >= 0) {
555 stack_emplace_back(node_1.child_l, id_s2);
556 stack_emplace_back(node_1.child_r, id_s2);
557 if (node_1.box_num > 0) {stack_emplace_back(negate(id_1), id_2);}
558 } else if (id_s2 >= 0) {
559 stack_emplace_back(id_s1, node_2.child_l);
560 stack_emplace_back(id_s1, node_2.child_r);
561 // FIXME if (node_2.box_num > 0) {stack_emplace_back(id_1, negate(id_2));}
562 }
563 }
564
565 // Return true if the trees intersect
566 return !candidates.empty();
567 }

593 template <typename Object>
594 Real distance(Object const & obj, IndexSet & candidates) const
595 {
596 // Reset statistics
597 this->m_check_counter = 0;
598
599 // Return a negative value if the tree is empty
600 if (this->is_empty() || this->is_empty()) {return -1.0;}
601
602 // Collect the original object boxes
603 BoxUniquePtrList const & boxes{*this->m_boxes};
604
605 // Setup the stack
606 this->m_stack.clear();
607 this->m_stack.reserve(2*this->size() + 1);
608 this->m_stack.emplace_back(0);
609 this->m_stack.emplace_back(0);
610
611 // Main loop that checks the intersection iteratively
612 Real distance{std::numeric_limits<Real>::max()};
613 candidates.clear();
614 while (!this->m_stack.empty())
615 {
616 // Pop the node from stack
617 Integer const id{this->m_stack.back()}; this->m_stack.pop_back();
618
619 // Get the node
620 Node const & node {this->m_tree_structure[id]};
621
622 // Compute the distance between the object and the bounding box
623 ++this->m_check_counter;
624 Real tmp_distance{node.box.interior_distance(obj)};
625
626 // If the distance is greater than the temporary minimum distance, skip the node
627 if (tmp_distance > distance) {continue;}
628
629 // Compute the distance between the object and the long boxes on the node
630 if (node.box_num > 0) {
631 ++this->m_check_counter;
632 if (node.box_long.interior_distance(obj) <= distance) {
633 Integer const id_ini{node.box_ptr};
634 Integer const id_end{node.box_ptr + node.box_num};
635 for (Integer i{id_ini}; i < id_end; ++i) {
636 Integer const pos{this->m_tree_boxes_map[i]};
637 ++this->m_check_counter;
638 tmp_distance = boxes[pos]->interior_distance(obj);
639 if (tmp_distance < distance) {
640 candidates.clear(); candidates.insert(pos); distance = tmp_distance;
641 } else if (tmp_distance == distance) {
642 candidates.insert(pos);
643 }
644 }
645 }
646 }
647
648 // Push children on the stack if thay are not leafs
649 if (node.child_l > 0) {this->m_stack.emplace_back(node.child_l);}
650 if (node.child_r > 0) {this->m_stack.emplace_back(node.child_r);}
651 }
652
653 // Return the distance between the point and the tree
654 return distance;
655 }

663 Real distance(Tree const & tree, IndexMap & candidates) const
664 {
665 // Reset statistics
666 this->m_check_counter = 0;
667
668 // Return if the tree is empty
669 if (this->is_empty() || tree.is_empty()) {return -1.0;}
670
671 // Collect the original object boxes
672 BoxUniquePtrList const & boxes_1{*this->m_boxes};
673 BoxUniquePtrList const & boxes_2{*tree.m_boxes};
674
675 // Setup the stack
676 this->m_stack.clear();
677 this->m_stack.reserve(this->size() + tree.size() + 2);
678 this->m_stack.emplace_back(0);
679 this->m_stack.emplace_back(0);
680
681 // Negate the id of the node to distinguish the tree
682 auto negate = [] (Integer const id) {return -1-id;};
683
684 // Main loop that checks the intersection iteratively
685 Real distance{std::numeric_limits<Real>::max()};
686 candidates.clear();
687 while (!this->m_stack.empty())
688 {
689 // Pop the node from stack (reversed order)
690 Integer const id_s2{this->m_stack.back()}; this->m_stack.pop_back();
691 Integer const id_2{id_s2 >= 0 ? id_s2 : negate(id_s2)};
692 Integer const id_s1{this->m_stack.back()}; this->m_stack.pop_back();
693 Integer const id_1{id_s1 >= 0 ? id_s1 : negate(id_s1)};
694
695 // Get the node
696 Node const & node_1{this->m_tree_structure[id_1]};
697 Node const & node_2{tree.m_tree_structure[id_2]};
698
699 // Compute the distance between the bounding boxes
700 ++this->m_check_counter;
701 Real tmp_distance{node_1.box.interior_distance(node_2.box)};
702
703 // If the distance is greater than the temporary minimum distance, skip the nodes
704 if (tmp_distance > distance) {continue;}
705
706 // Compute the distance between the long boxes on the nodes
707 if (node_1.box_num > 0 && node_2.box_num > 0) {
708 ++this->m_check_counter;
709 if (node_1.box_long.interior_distance(node_2.box_long) <= distance) {
710 Integer const id_1_ini{node_1.box_ptr};
711 Integer const id_1_end{node_1.box_ptr + node_1.box_num};
712 Integer const id_2_ini{node_2.box_ptr};
713 Integer const id_2_end{node_2.box_ptr + node_2.box_num};
714 for (Integer i{id_1_ini}; i < id_1_end; ++i) {
715 Integer const pos_1{this->m_tree_boxes_map[i]};
716 for (Integer j{id_2_ini}; j < id_2_end; ++j) {
717 Integer const pos_2{tree.m_tree_boxes_map[j]};
718 ++this->m_check_counter;
719 tmp_distance = boxes_1[pos_1]->interior_distance(*boxes_2[pos_2]);

769 template <typename Function = std::function<Real(Point const &, Box const &)>>
770 Real closest(Point const & point, Integer const n, IndexSet & candidates, Function
771 distance_function = [] (Point const & p, Box const & b) {return b.interior_distance(p);}) const
772 {
773 // Reset statistics
774 this->m_check_counter = 0;
775
776 // Return if the tree is empty
777 if (this->is_empty()) {return -1.0;}
778
779 // Collect the original object boxes
780 BoxUniquePtrList const & boxes{*this->m_boxes};
781
782 // Setup the stack
783 this->m_stack.clear();
784 this->m_stack.reserve(2*this->size() + 1);
785 this->m_stack.emplace_back(0);
786
787 // Candidate vector distance and index
788 using Pair = std::pair<Real, Integer>;
789 auto cmp = [](const Pair & a, const Pair & b) {return a.first < b.first;};
790 std::priority_queue<Pair, std::vector<Pair>, decltype(cmp)> queue(cmp);
791
792 // Main loop that checks the intersection iteratively
793 Real max_distance{std::numeric_limits<Real>::max()};
794 candidates.clear();
795 while (!this->m_stack.empty())
796 {
797 // Pop the node from stack
798 Integer const id{this->m_stack.back()}; this->m_stack.pop_back();
799
800 // Get the node
801 Node const & node{this->m_tree_structure[id]};
802
803 // Compute the distance between the point and the bounding box
804 ++this->m_check_counter;
805 Real tmp_distance{distance_function(point, node.box)};
806
807 // If the distance is greater than the maximum distance, skip the node
808 if (tmp_distance > max_distance) {continue;}
809
810 // Compute the distance between the point and the long boxes on the node
811 if (node.box_num > 0) {
812 Integer const id_ini{node.box_ptr};
813 Integer const id_end{node.box_ptr + node.box_num};
814 for (Integer i{id_ini}; i < id_end; ++i) {
815 Integer const pos{this->m_tree_boxes_map[i]};
816 ++this->m_check_counter;
817 tmp_distance = distance_function(point, *boxes[pos]);
818 if (tmp_distance < max_distance) {
819 max_distance = tmp_distance;
820 if (static_cast<Integer>(queue.size()) < n) {
821 queue.emplace(tmp_distance, pos);
822 } else if (tmp_distance < queue.top().first) {
823 queue.pop();
824 queue.emplace(tmp_distance, pos);
825 }
826 }
827 }
828 }
829
830 // Push children on the stack if thay are not leafs
831 if (node.child_l > 0) {this->m_stack.emplace_back(node.child_l);}
832 if (node.child_r > 0) {this->m_stack.emplace_back(node.child_r);}
833 }
834
835 // Extract indices into candidates
836 while (!queue.empty()) {candidates.insert(queue.top().second); queue.pop();}
837 return max_distance;
838 }
